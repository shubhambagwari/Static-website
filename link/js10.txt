	var survey=new Array(4);
	for(i=0;i<4;i++)
		survey[i]=new Array(4);
	for(i=0;i<4;i++)
	{
	  for(j=0;j<4;j++)
		survey[i][j]=0;
	}

function storeArray()
{
	var citycode=document.getElementById("city").value;
	var carcode=parseInt(document.getElementById("car").value);
	
	
		var num = survey[citycode][carcode];
		num = num+1;
		survey[citycode][carcode] = num;
	var c=1;
	for(i=0;i<4;i++)
	{
		
		for(j=0;j<4;j++)
		{
			document.getElementById("c"+c).innerHTML=survey[i][j];
			c++;
		}
		
	}

}
