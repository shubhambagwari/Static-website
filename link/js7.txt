<script>
function validate()
{
	var x = document.getElementById("phone").value;
	if(x.length<10)
	document.getElementById("show").innerHTML = "* Invalid Input";
	else
	document.getElementById("show").innerHTML = "";
}

function check()
{	
	var y = document.getElementById("phone").value;
	if(y.length==10)
	{
		var f1 = y.substr(0,3);
		document.getElementById("abc").innerHTML = f1;
		var f1 = y.substr(3,10);
		document.getElementById("last").innerHTML = f1;
	}
	else
	{
		document.getElementById("abc").innerHTML = "";
		document.getElementById("last").innerHTML = "";	
	}
}
</script>
