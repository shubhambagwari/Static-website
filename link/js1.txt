	<script>
		function Master_valid()
		{
			var x=document.getElementById("mast").value;
			var patt=/^(51|52|53|54|55)[0-9]/;
			if(!patt.test(x) || x.length!=16)
				document.getElementById("error1").innerHTML="Invalid Input";
			else
				document.getElementById("error1").innerHTML="valid ";
		}
		function Visa_valid()
		{
			var x=document.getElementById("visa").value;
			var patt=/^4[0-9]/;
			if(!patt.test(x) || (x.length!=12 && x.length!=15))
				document.getElementById("error2").innerHTML="Invalid Input";
			else
				document.getElementById("error2").innerHTML="valid ";
		}
		function AmExp_valid()
		{
			var x=document.getElementById("exp").value;
			var patt=/^(34|37)[0-9]/;
			if(!patt.test(x) || x.length!=15)
				document.getElementById("error3").innerHTML="Invalid Input";
			else
				document.getElementById("error3").innerHTML="valid ";
		}
